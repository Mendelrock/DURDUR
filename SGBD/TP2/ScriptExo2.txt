REM Exercice 2

drop table Chapitre purge;
drop table livre purge;

create table Chapitre (
        IdChap integer not null,
        Titre varchar(30) not null,
        livre_id integer,
        primary key (idChap, livre_id));

create table Livre (
        IdLivre integer not null,
        Libelle varchar(30) not null,
        primary key (idLivre));

alter table Chapitre 
        add constraint FK_Chapitre_Livre
        foreign key (livre_id) 
        references Livre(IdLivre) ON DELETE CASCADE;
